# ETL Process - Add FeeRules Table and Load Data Into Student Table

**Overview**

This part is mainly focus on ETL part which is Extract data from excel file, transform the data and load it into databases.

import pandas as pd

## Insert Data Into Database Using Python Code

```json
import pandas as pd
import psycopg2
from datetime import datetime

# =========================
# 1️⃣ Load cleaned CSV
# =========================
df = pd.read_csv("cleaned_students.csv", delimiter='|')

# =========================
# 2️⃣ Clean / convert columns
# =========================
text_columns = ['Address', 'Area', 'School', 'FatherJob', 'MotherJob', 'Phone']
df[text_columns] = df[text_columns].fillna('')

numeric_columns_int = ['SiblingsCount']
for col in numeric_columns_int:
    df[col] = pd.to_numeric(df[col], errors='coerce').fillna(0).astype(int)

numeric_columns_float = ['FamilyIncome', 'Fee']
for col in numeric_columns_float:
    df[col] = pd.to_numeric(df[col], errors='coerce').fillna(0.0)

# Ensure IC_clean is 12-digit string
df['IC_clean'] = df['IC_clean'].fillna('000000000000')
df['IC_clean'] = df['IC_clean'].astype(str).str.replace(r'[^0-9]', '', regex=True)
df['IC_clean'] = df['IC_clean'].apply(lambda x: x.zfill(12))

# =========================
# 3️⃣ Connect to PostgreSQL
# =========================
conn = psycopg2.connect(
    host="db.ulidbafyhvehmdmzmuxq.supabase.co",
    dbname="postgres",
    user="postgres",
    password="mypassword",
    port=5432
)
cur = conn.cursor()

# =========================
# 4️⃣ Insert rows one by one
# =========================
insert_query = """
INSERT INTO "Students"
("Name", "IcNumber", "ClassId", "Year", "Address", "Area", "PhoneNumber",
 "School", "FatherJob", "MotherJob", "FamilyIncome", "SiblingsCount", "CreatedAt")
VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s)
"""

rows_inserted = 0

for _, row in df.iterrows():
    cur.execute(insert_query, (
        row['Name'],
        row['IC_clean'],
        1,                      # ClassId
        2025,                   # Year
        row['Address'],
        row['Area'],
        row['Phone'],
        row['School'],
        row['FatherJob'],
        row['MotherJob'],
        row['FamilyIncome'],
        row['SiblingsCount'],
        datetime.now()
    ))
    rows_inserted += 1

conn.commit()
print(f"✅ {rows_inserted} rows inserted successfully!")

# =========================
# 5️⃣ Close connection
# =========================
cur.close()
conn.close()
```
